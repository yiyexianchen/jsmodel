点击两个按钮切换div 

                        $('#depmap').click(function(e){
                            $(".row.dep").css("display","none");
                            $("#depmap").addClass("active");
                            $("#deptree").removeClass("active");
                            $(".row.staff").css("display","block");
                        });

                        $('#deptree').click(function(e){
                            $(".row.dep").css("display","block");
                            $("#depmap").removeClass("active");
                            $("#deptree").addClass("active");
                            $(".row.staff").css("display","none");
                        });
